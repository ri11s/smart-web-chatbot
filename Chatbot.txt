  <script>
    const backgrounds = [
      "var(--bg-1)",
      "var(--bg-2)",
      "var(--bg-3)",
      "var(--bg-4)"
    ];
    let currentBg = 0;

    setInterval(() => {
      currentBg = (currentBg + 1) % backgrounds.length;
      document.body.style.background = backgrounds[currentBg];
    }, 15000); // كل 15 ثانية

    function sendMessage() {
      const input = document.getElementById("userInput");
      const chat = document.getElementById("chat");
      const msg = input.value.trim();

      if (msg === "") return;

      addMessage(msg, "user");
      const reply = getSmartReply(msg);
      addMessage(reply, "bot");

      input.value = "";
      chat.scrollTop = chat.scrollHeight;
    }

    function addMessage(text, sender) {
      const chat = document.getElementById("chat");
      const div = document.createElement("div");
      div.className = "message " + sender;
      div.textContent = text;
      chat.appendChild(div);
    }

    function getSmartReply(text) {
      const msg = text.toLowerCase();

      if (msg.includes("سلام") || msg.includes("السلام"))
        return "مرحبًا، كيف يمكنني مساعدتك؟";

      if (msg.includes("مين انت؟") || msg.includes("من انت"))
return "أنا مساعد ذكي مخصص للرد على الأسئلة العامة والمهنية.";

      if (msg.includes("برمجة") || msg.includes("أتعلم"))
        return "هل تبحث عن مصادر لتعلم البرمجة أم نصائح لتحديد المسار المهني؟";

      if (msg.includes("تقدر تساعدني في تخصصي؟") || msg.includes("هندسة"))
        return "هل يمكنك توضيح سؤالك المتعلق بتخصصك؟ هل تبحث عن مجالات عمل أو مشاريع تطبيقية؟";

      if (msg.includes("مشروع") || msg.includes("فكرة"))
        return "هل ترغب بمقترحات لمشاريع أكاديمية أم عملية مرتبطة بتخصصك؟";

      if (msg.includes("بايثون") || msg.includes("python"))
        return "هل تبحث عن تطبيقات معينة بلغة Python أو ترغب بتوصيات لمكتبات تستخدمها؟";

      if (msg.includes("ويب") || msg.includes("web"))
        return "هل تهتم بتطوير الواجهات الأمامية (Frontend) أم الخلفية (Backend)؟";

      if (msg.includes("سيرة ذاتية") || msg.includes("cv"))
        return "هل ترغب بمساعدة في كتابة السيرة الذاتية أم تنسيقها؟";

      return "يرجى توضيح السؤال بشكل أدق حتى أتمكن من مساعدتك.";
    }
